//Portas analógicas: São portas de entrada

#define led 3
#define pot A0

void setup() 
{
  pinMode (led,OUTPUT);
  pinMode (pot, INPUT);  

  Serial.begin(9600);
}

void loop() 
{
  //Analogico = 0 a 1023
  //PWM = 0 a 255

  //analogWrite(led, analogRead(pot)/4 );
  analogWrite(led, map(analogRead(pot),0,1023,0,255));
  //Sintaxe = map(leitura, min_val_entrada, max_val_entrada, min_val_saida, max_val_saida);
    
}
